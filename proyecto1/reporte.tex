\chapter{Introducción}
\noindent
Definimos la Gramática para \large M\small INI\large L\small ISP \normalsize en \textbf{EBNF}:

\begin{tcolorbox}[colback=azulin!5!white, colframe=azulin!80, title=Gramática MINILISP]
  \[
  \renewcommand{\arraystretch}{1.1}
  \begin{array}{rcl}
    Expr &::=& Var \\
    &\mid& Num \\
    &\mid& Bool \\
    &\mid& (+\ Expr\ \{Expr\}) \\
    &\mid& (-\ Expr\ \{Expr\}) \\
    &\mid& (*\ Expr\ \{Expr\}) \\
    &\mid& (/\ Expr\ \{Expr\}) \\
    &\mid& (add1\ Expr) \\
    &\mid& (sub1\ Expr) \\
    &\mid& (sqrt\ Expr) \\
    &\mid& (expt\ Expr) \\
    &\mid& (not\ Expr) \\
    &\mid& (=\ Expr\ \{Expr\}) \\
    &\mid& (<\ Expr\ \{Expr\}) \\
    &\mid& (>\ Expr\ \{Expr\}) \\
    &\mid& (<=\ Expr\ \{Expr\}) \\
    &\mid& (>=\ Expr\ \{Expr\}) \\
    &\mid& (!=\ Expr\ \{Expr\}) \\
    &\mid& (Expr,\ Expr) \\
    &\mid& (fst\ Expr) \\
    &\mid& (snd\ Expr) \\
    &\mid& (let\ (Var\ Expr)\ \{Expr\}) \\
    &\mid& (letrec\ (Var\ Expr)\ \{Expr\}) \\
    &\mid& (let*\ (Var\ Expr)\ \{Expr\}) \\
    &\mid& (if0\ Expr\ Expr\ Expr) \\
    &\mid& (if\ Expr\ Expr\ Expr) \\ 
    &\mid& (lambda\ (\{Var\})\ \{Expr\}) \\
    &\mid& (Expr\ Expr) \\
    &\mid& [\ \{Expr,\}\ ] \\
    &\mid& (head\ Expr) \\
    &\mid& (tail\ Expr) \\
    &\mid& (cond\ [Expr\ Expr]\ \{[Expr\ Expr]\}\ [else\ Expr]) \\
    Var &::=& \textit{Identificador de variable} \\
    Num &::=& \textit{Constante entera} \\
    Bool &::=& \#t \mid \#f
  \end{array}
  \]
\end{tcolorbox}
